## 3. MockLLMClient 设计

```python
# tests/e2e/mock_llm.py
class MockLLMClient:
    """Mock LLM——确定性输出，可控失败。

    实现与 LLMClient 相同接口（generate 方法），
    测试通过构造函数注入 Scheduler。
    """

    def __init__(
        self,
        fail_count: int = 0,        # 前 N 次调用拋异常（模拟 5xx）
        fixed_response: str = "mock response",
        entropy: float = 0.0,       # >0.75 触发 L3 告警
    ):
        self.fail_count = fail_count
        self.fixed_response = fixed_response
        self.entropy = entropy
        self.call_count = 0

    async def generate(self, request: LLMRequest, task_id: str) -> LLMResponse:
        self.call_count += 1
        if self.call_count <= self.fail_count:
            raise LLMError(500, "Mock LLM 5xx")
        return LLMResponse(
            content=self.fixed_response,
            model="mock-model",
            usage=LLMUsage(
                prompt_tokens=10,
                completion_tokens=20,
                total_tokens=30,
            ),
            degraded=False,
        )
```

## 4. ProcessSandbox 设计（L2 兜底）

### 4.1 接口——与 Sandbox 同签名

```python
# src/orbit/sandbox/process_sandbox.py
class ProcessSandbox:
    """进程级沙箱——Docker 不可用时的兜底。

    实现与 Sandbox 相同的 run(code, language) → str 接口，
    调度器不感知底层机制差异。
    """

    async def run(self, code: str, language: str = "python") -> str:
        self._check_available()  # 不可用时抛 SandboxError
        if sys.platform == "win32":
            return await self._run_appcontainer(code)
        else:
            return await self._run_unix_sandbox(code)

    async def is_available(self) -> bool:
        # Windows: 检测 AppContainer 创建权限
        # Unix: 检测 subprocess + rlimit
```

### 4.2 Windows 实现——AppContainer + Job + LowIL

```python
async def _run_appcontainer(self, code: str) -> str:
    """三层 Windows 原生隔离。

    用 pywin32 调用 Win32 API：
    - CreateAppContainerProfile("OrbitSandbox") → capability SID
    - SetNamedSecurityInfo(temp_dir, DACL, appcontainer_sid) → 授权工作目录
    - CreateProcess(lpProcessAttributes=SECURITY_CAPABILITIES)
    - AssignProcessToJobObject(job) → 资源限制
    - SetTokenInformation(LowIL) → 写保护
    - WaitForSingleObject(timeout) → 超时 TerminateProcess
    """
```

**依赖**：`pywin32>=306`（Windows-only dep，`sys.platform == "win32"` 时 import）。

### 4.3 Unix 实现

```python
async def _run_unix_sandbox(self, code: str) -> str:
    proc = await asyncio.create_subprocess_exec(
        "python", "-c", code,
        cwd=temp_dir,
        env={"PATH": "/usr/bin", "HOME": temp_dir},
        preexec_fn=self._drop_privileges,
        stdout=PIPE, stderr=PIPE,
    )
    return (await asyncio.wait_for(proc.communicate(), timeout)).stdout.decode()
```

### 4.4 沙箱工厂——硬停止规则

```python
# src/orbit/sandbox/sandbox_factory.py
async def create_sandbox() -> Sandbox | ProcessSandbox:
    """按优先级选择沙箱。

    Docker → ProcessSandbox → 拒绝启动。
    """
    docker = Sandbox()
    if await docker.is_available():
        logger.info("sandbox_mode", mode="Docker")
        return docker

    process = ProcessSandbox()
    if await process.is_available():
        logger.warning("sandbox_mode", mode="ProcessSandbox", note="降级运行")
        return process

    raise RuntimeError(
        "无可用沙箱隔离机制。"
        "请安装 Docker 或启用 Windows AppContainer。"
        "系统无法在不安全环境运行。"
    )
```

## 6. E2E Fixture 设计

```python
# tests/e2e/conftest.py

@pytest.fixture(scope="session")
def docker_available() -> bool:
    """检测 Docker 是否可用——不可用时 PG/Redis 走 SQLite/fakeredis fallback。"""
    import subprocess
    try:
        subprocess.run(["docker", "info"], capture_output=True, timeout=5, check=True)
        return True
    except Exception:
        return False

@pytest.fixture(scope="session")
async def e2e_app(docker_available: bool):
    """启动 E2E 应用——含真实 scheduler + mock LLM + 真实沙箱。

    scope=session：所有 E2E 测试共享同一 app 实例，加速执行。
    """
    # 启动 infra
    if docker_available:
        subprocess.run(
            ["docker", "compose", "-f", "docker-compose.test.yml", "up", "-d", "--wait"],
            check=True,
        )
        db_url = "postgresql://test:test@localhost:5433/orbit_test"
        redis_url = "redis://localhost:6380/0"
    else:
        db_url = "sqlite:///tests/e2e/test.db"  # fallback
        redis_url = None

    # 组装 app
    bus = EventBus()
    llm = MockLLMClient()
    sandbox = await create_sandbox()  # 自动选择 Docker/Process/拒绝
    cp = CheckpointManager(db_url=db_url, redis_url=redis_url)
    scheduler = Scheduler(llm_client=llm, checkpoint_manager=cp, event_bus=bus)
    app = create_app(event_bus=bus)
    app.state.scheduler = scheduler

    async with httpx.AsyncClient(app=app, base_url="http://test") as client:
        yield client

    # teardown
    if docker_available:
        subprocess.run(
            ["docker", "compose", "-f", "docker-compose.test.yml", "down", "-v"],
            check=True,
        )
    else:
        Path("tests/e2e/test.db").unlink(missing_ok=True)
```

## 10. 风险点

| 风险 | 缓解 |
|------|------|
| pywin32 在 CI Linux 环境不可用 | `sys.platform == "win32"` 条件导入，CI 走 Docker 或 Unix 兜底 |
| E2E 执行时长超 CI 限制 | 5 场景 8 用例 ≤2 分钟（Mock LLM 无网络延迟），设 timeout=120 |
| docker-compose 端口冲突 | 专用端口 5433/6380，CI jobs 隔离 |
| PG 容器启动慢于测试 | `--wait` flag + healthcheck 探针 |
