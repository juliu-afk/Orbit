> 覆盖 SC1-SC3 · 无偏离

## 1. docker-compose.vlm.yml

```yaml
services:
  minicpm-v:
    image: vllm/vllm-openai:latest
    ports: ["8000:8000"]
    volumes: ["~/.cache/huggingface:/root/.cache/huggingface"]
    command: --model openbmb/MiniCPM-V-4.6 --max-model-len 4096
    deploy:
      resources:
        reservations:
          devices: [{driver: nvidia, count: 1, capabilities: [gpu]}]
```

## 2. Tier.PRIVATE 路由

```python
# multimodal.py 新增
Tier.PRIVATE = 0  # 本地模型

TIERS[Tier.PRIVATE] = TierConfig(
    model="openbmb/MiniCPM-V-4.6",
    endpoint="http://localhost:8000/v1",
    thinking=None,  # MiniCPM 无 thinking
    input_cost_per_million=0.0,
    output_cost_per_million=0.0,  # 本地=免费
    context_window=262_000,
    description="P0 私有化：数据不出域，6GB VRAM",
)

# 隐私关键词
PRIVATE_KEYWORDS = ["财务", "内部", "密钥", "密码", "工资", "对账单"]

# TierRouter.classify() 加隐私检测
if any(kw in all_text for kw in PRIVATE_KEYWORDS):
    return Tier.PRIVATE
```
