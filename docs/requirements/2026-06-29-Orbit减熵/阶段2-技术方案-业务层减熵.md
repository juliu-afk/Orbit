```python
from __future__ import annotations
from dataclasses import dataclass
from tree_sitter import Language, Parser, Node  # 已有依赖

@dataclass
class CodeFragment:
    """提取出的代码片段"""
    identifier: str         # 函数名/类名
    node_type: str          # "function_definition" / "class_definition"
    source: str             # 源代码
    line_start: int
    line_end: int
    relevance_score: float  # 0-1

class RelevanceScorer:
    """基于关键词匹配 + AST 结构的关系打分"""

    def __init__(self, parser: Parser):
        self._parser = parser

    def score(
        self, code: str, keywords: list[str], max_fragments: int = 5
    ) -> list[CodeFragment]:
        """
        解析代码为 AST → 提取所有函数/类定义 →
        对每个定义计算与关键词的相关性 → 排序 → 返回 top-N

        打分规则：
        - 标识符直接包含关键词：+0.5
        - 标识符与关键词相似（jaro_winkler > 0.8）：+0.3
        - 函数体内出现关键词：+0.2
        - 被调用关系（调用者出现在搜索结果中）：+0.1

        tree = self._parser.parse(bytes(code, "utf-8"))
        fragments: list[CodeFragment] = []

        # 遍历 AST 提取所有定义节点
        self._extract_definitions(
            tree.root_node, code, keywords, fragments
        )

        # 按相关性降序排序
        fragments.sort(key=lambda f: f.relevance_score, reverse=True)
```

```python
"""Agent 任务 → 模板匹配"""

class TemplateSelector:
    """根据任务描述匹配最佳代码模板"""

    def __init__(self, store: KnowledgeStore):
        self._store = store

    async def select(
        self, task_description: str
    ) -> list[TemplateMatch]:
        """
        1. 从任务描述提取关键词（"新增 API" / "写测试" / "Service 类"）
        2. 搜索 MANIFEST.yaml 的 applicable_when 匹配
        3. 返回匹配的模板（按置信度排序）
        """
        keywords = extract_keywords(task_description)
        matches = []

        for template in self._store.list_templates():
            score = self._match_score(template, keywords)
            if score > 0.5:
                matches.append(TemplateMatch(template=template, score=score))

        matches.sort(key=lambda m: m.score, reverse=True)
        return matches
```

### 2.4 编辑摇摆检测（P1）

**检测逻辑**：

```python
# scheduler/edit_stability.py（新文件）
"""文件编辑稳定性检测——高熵文件触发需求重确认"""

from datetime import datetime, timedelta
from collections import Counter

class EditStabilityDetector:
    """基于代码图谱的编辑摇摆检测"""
    HIGH_ENTROPY_THRESHOLD = 4       # 7 天内 ≥4 次编辑
    PING_PONG_THRESHOLD = 2          # 同样逻辑改回 2 次
    AGENT_CONFLICT_THRESHOLD = 3     # ≥3 个 Agent 编辑同一文件

    def __init__(self, code_graph: CodeGraph, event_bus: EventBus):
        self._graph = code_graph
        self._bus = event_bus

    async def on_file_changed(self, event: FileChangedEvent) -> None:
        """文件变更事件 → 检测摇摆模式"""
        history = await self._graph.get_file_history(
            event.file_path, days=7
        )

        if len(history) < self.HIGH_ENTROPY_THRESHOLD:
            return  # 编辑次数还不够

        issues: list[str] = []

        # 检测 1：编辑次数过多
        if len(history) >= self.HIGH_ENTROPY_THRESHOLD:
            issues.append(
                f"文件 {event.file_path} 近 7 天编辑 {len(history)} 次"
            )

        # 检测 2：回弹模式（改为 A → 改回 B → 再改 A）
        functions = self._extract_changed_functions(history)
        for func_name, versions in functions.items():
            if self._has_ping_pong(versions):
                issues.append(
                    f"函数 {func_name} 存在回弹模式——"
                    "相同逻辑反复修改"
                )

        # 检测 3：多 Agent 竞争
        agents = Counter(h.agent_id for h in history)
        if len(agents) >= self.AGENT_CONFLICT_THRESHOLD:
            issues.append(
                f"文件被 {len(agents)} 个 Agent 编辑——可能信息不同步"
            )

        if issues:
            await self._trigger_clarification(event.file_path, issues)
```

### 2.6 依赖膨胀拦截（P2）

**拦截逻辑**：

```python
# resource_guard/dependency_guard.py（新文件）
"""Agent 提议安装新包时检查已有替代方案"""

class DependencyGuard:
    """依赖安装拦截——防止膨胀"""

    def __init__(self, code_graph: CodeGraph):
        self._code_graph = code_graph

    async def check(self, proposed_package: str, reason: str) -> DependencyCheck:
        """
        检查提议的新依赖：
        1. 标准库有没有替代？
        2. 已有依赖有没有替代？
        3. 新依赖有几个传递依赖？
        4. 包的活跃度和许可证？
        """
        result = DependencyCheck(package=proposed_package)

        # 检查 1：已有标准库替代
        std_alt = STD_LIB_ALTERNATIVES.get(proposed_package)
        if std_alt:
            result.stdlib_alternative = std_alt
            result.recommendation = f"建议用标准库 {std_alt} 替代 {proposed_package}"

        # 检查 2：已有项目的替代
        existing = await self._code_graph.find_similar_package(proposed_package)
        if existing:
            result.existing_alternative = existing
            result.recommendation = f"项目已有 {existing}，考虑复用"

        # 检查 3：传递依赖
        transitive = await self._get_transitive_deps(proposed_package)
        result.transitive_deps = transitive
        if len(transitive) > 5:
            result.warnings.append(f"将新增 {len(transitive)} 个传递依赖")

        # 检查 4：活跃度/许可证（通过 PyPI API）
        pypi_info = await self._fetch_pypi_info(proposed_package)
        if pypi_info:
            result.last_release = pypi_info.get("release_date")
            result.license = pypi_info.get("license")

        result.needs_confirmation = (
            result.stdlib_alternative is not None
            or result.existing_alternative is not None
            or len(transitive) > 5
        )

        return result

# 常见"可被标准库替代"的包
```

## 三、数据流

```
用户任务
    ↓
TemplateSelector → 搜模板 ───→ 命中:注入模板
    ↓                           未命中:自由生成
RelevanceScorer → 裁剪上下文
    ↓
PromptBuilder → 注入决策日志 + RULES_BLOCK(含简洁规则)
    ↓
Agent 生成代码
    ↓
ConcisenessGuard → 简洁审计（>200行）
    ↓
DependencyGuard → 依赖检查（pip install 拦截）
    ↓
EditStabilityDetector → 摇摆检测（文件变更事件）
    ↓
TestGapDetector → 空洞检测（QA Agent 调用）
    ↓
ClaudeMdGenerator → 项目说明书更新（图谱变更触发）
```

## 五、防幻觉层影响

| 策略 | 影响层 | 判定逻辑 | 误报风险 |
|------|-------|---------|---------|
| 上下文裁剪 | L2 动态追踪 | 更精确的上下文意味着更少的幻觉——正向影响 | 低——裁剪过激可能导致 Agent 缺少上下文，但降级策略兜底 |
| 简洁约束 | 全部 | Agent 输出更短——防幻觉层分析量减少 | 低——不改变判定逻辑，只改变输入 |
| 模板库 | L1 静态校验 | 模板代码已经过静态校验——模板生成的代码天然合规 | 极低 |
| 依赖拦截 | L8 配置漂移 | 阻止不必要的新依赖 → 减少配置漂移源 | 低——误拦时用户手动确认 |

## 七、边界 case 清单

| 类别 | 场景 | 预期行为 |
|------|------|---------|
| 上下文裁剪 | 任务关键词匹配不到任何函数 | 退回到全文硬截断（5000 字符） |
| 上下文裁剪 | 代码文件有语法错误导致 Tree-sitter 解析失败 | 退回到全文注入 + 标注"解析失败" |
| 简洁约束 | Agent 确实需要 300 行才能表达清楚 | 阈值触发 Reviewer 审计——Reviewer 判断"合理"则放行 |
| 简洁约束 | Reviewer 和 Developer Agent 意见冲突 | 以用户判断为准——输出提示等待确认 |
| 模板库 | 模板命中但生成后跑不过测试 | Agent 回退到自由生成——模板标记为"需修正" |
| 模板库 | MANIFEST.yaml 描述与实际模板不一致 | CI 中有模板验证脚本——yaml 与 .j2 的一致性检查 |
| 摇摆检测 | 项目初期（<7 天）无足够历史数据 | 渐进式——1 天起用，阈值随数据量动态调整 |
| 摇摆检测 | 正当的频繁修改（用户反复调整需求） | 标记为"高熵但可能合理"，不阻断但提示 |
| 决策日志 | 旧决策与新需求矛盾 | 标记冲突 → Clarifier 介入 |
| 依赖拦截 | pip install 不在拦截列表里（新包） | 放行，异步分析后补充到已知替代列表 |
| 空洞检测 | 参数无类型注解 | 跳过该参数，不报"无类型"（因为无法分析） |
| CLAUDE.md 生成 | 生成的 CLAUDE.md 与人工编辑部分格式冲突 | 自动生成部分有 `<!--- auto-generated --->` 标记，人工部分不受影响 |
| 全部 | 用户关闭某个策略（配置开关） | 该策略完全跳过，不增加任何 overhead |

## 八、风险与缓解

| 风险 | 概率 | 影响 | 缓解 |
|------|------|------|------|
| 上下文裁剪过激导致 Agent 信息不足 | 中 | 高——错误决策 | 降级策略 + 相关性阈值可调 |
| 简洁约束压缩了必要的注释/可读性 | 中 | 中——代码难维护 | 阈值触发 Reviewer 介入，不是硬截断 |
| 模板库质量不够导致生成代码风格差 | 低 | 中——用户不信任 | 从 Orbit 自身代码提取模板（已验证） |
| 摇摆检测误报导致不必要的中断 | 中 | 低——多一次澄清 | 阈值可调，渐进启用 |
| 依赖拦截数据库覆盖面不够 | 中 | 低——漏放 | 初版手动维护常见包列表，后续社区贡献 |
