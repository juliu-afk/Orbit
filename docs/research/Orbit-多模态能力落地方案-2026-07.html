# src/orbit/gateway/multimodal.py
from dataclasses import dataclass
from enum import Enum

class MediaType(Enum):
    VIDEO = "video"
    IMAGE = "image"
    DOCUMENT = "document"
    GUI = "gui"

@dataclass
class RouteConfig:
    primary: str       # model name
    fallback: str
    max_cost: float    # max budget per call (yuan)

ROUTES = {
    MediaType.VIDEO: RouteConfig(
        primary="glm-4.1v-thinking-flash",     # 免费，原生 2h
        fallback="qwen3.7-plus",               # 1M ctx, MLVU #1
        max_cost=0,
    ),
    MediaType.IMAGE: RouteConfig(
        primary="qwen3.7-plus",                # Vision Arena 国内第1
        fallback="glm-4.1v-thinking-flash",    # 免费
        max_cost=0.01,
    ),
    MediaType.DOCUMENT: RouteConfig(
        primary="deepseek-ocr-2",              # $0.15/M 均一
        fallback="qwen-vl-ocr",
        max_cost=0.005,

        primary="glm-4.1v-thinking-flash",     # 原生 UI2Code
        fallback="qwen3.7-plus",               # ScreenSpot 79.0

}

📋 Part D：必要性说明

🔴 不做的代价

  • Agent 盲视：只能读代码和日志，无法理解截图、录屏、UI——软件开发 80% 的信息载体
  • Bug 诊断低效：用户只能文字描述问题，agent 无法直接看录屏定位
  • Computer Use 缺失：没有 GUI Agent，无法做端到端自动化测试——只能调 API，不能操作真实 UI
  • 审计证据单薄：只有文本日志，缺少截图证据
  • 竞品降维打击：Codex/Cursor 已深度集成视觉理解，Orbit 不跟进则差异化为负

🟢 做的好处

  • 差异化护城河：唯一自研 Computer Use 的多 Agent 系统——Agent 能操作真实桌面软件
  • Bug 诊断提效 10×：录屏 → QA agent 自动分析 → 定位代码行
  • 审计合规：agent 每一步决策都有截图证据
  • 多模态知识摄入：技术视频 → 自动提取知识点 → Knowledge Graph
  • 国产自主可控：全链路国产模型，零外国闭源 API 依赖
