```python
# checkpoint/deviation.py — 偏离日志数据模型
from pydantic import BaseModel, Field
from enum import StrEnum

class DeviationSeverity(StrEnum):
    MAJOR = "major"        # 改变架构/数据模型/API 签名
    CRITICAL = "critical"  # 改变安全策略/回滚路径/熔断阈值

class DeviationEntry(BaseModel):
    timestamp: str                          # ISO 时间戳
    planned: str                            # 原计划做什么
    actual: str                             # 实际做了什么
    reason: str                             # 为什么偏离
    alternatives: list[str] = Field(...)    # 考虑的替代方案
    severity: DeviationSeverity
    file_refs: list[str] = Field(...)       # 涉及文件路径
    agent_id: str = ""                      # 哪个 Agent 记录的
```

```python
class DeviationLogger:
    """偏离日志记录器——Agent 执行循环中调用。
    
    WHY 独立类: 解耦记录逻辑与存储逻辑。
    CheckpointManager 管存储，DeviationLogger 管"何时算偏离"的判断。

    def __init__(self, task_id: str, plan_md: str = ""):

    def record(
        self, planned: str, actual: str, reason: str,
        alternatives: list[str], severity: DeviationSeverity,
        file_refs: list[str], agent_id: str = "",
    ) -> DeviationEntry:
        """记录一条偏离。返回创建的条目。"""
        ...
    
    async def flush_to_checkpoint(self, ckpt: CheckpointManager) -> None:
        """任务终止时写入 checkpoint。"""
        ...
    
    def render_markdown(self) -> str:
        """生成人类可读的 implementation-notes.md。"""
        ...
```

### 2.3 集成点——Agent 执行循环

偏离日志的调用点在 Agent 执行循环中。当前 Agent 循环在 `agents/base.py` 或 `loop/` 中。
在 Agent 决策分叉处插入：

```python
# 伪代码——Agent 循环中的调用点
# 当 Agent 检测到必须偏离原计划时：
if actual_action != planned_action and is_major_deviation(actual_action, planned_action):
    deviation_logger.record(
        planned=planned_action,
        actual=actual_action,
        reason="代码库边界条件: xxx",
        alternatives=["方案A", "方案B"],
        severity=DeviationSeverity.MAJOR,
        file_refs=["src/xxx.py"],
        agent_id=self.agent_id,
    )
```

**偏离判定逻辑**（`is_major_deviation`）:
- 数据模型变更（新增/修改/删除字段）→ MAJOR
- API 签名变更（路径/参数/响应结构）→ MAJOR
- 安全相关决策变更 → CRITICAL
- 文件命名/import 路径/变量命名 → 不算偏离

### 2.4 数据流

```
Agent 执行循环 → DeviationLogger.record()
    → 内存缓存（entries list）
    → 任务终止时 flush_to_checkpoint()
        → CheckpointManager.save() → Redis + PG
    → 同时写 implementation-notes.md（文件系统）
```

```python
class BlindspotReport(BaseModel):
    """盲区扫描报告。"""
    knowledge_gaps: list[str]     # 知识图谱缺失的概念
    code_gaps: list[str]          # 代码图谱未覆盖的模块/路径
    fuzzy_decisions: list[str]    # 需要用户确认的模糊决策点
    scan_duration_ms: float = 0.0
    scanned_modules: int = 0

class BlindspotScanner:
    """前向盲区扫描器——任务创建时执行。
    
    WHY 只读: 不修改任何图谱数据，仅查询。5 秒内完成。
    """
    
    def __init__(
        self,

        code_graph,          # CodeGraph 实例
    ):

        self._cg = code_graph
    
    async def scan(self, task_description: str, task_files: list[str]) -> BlindspotReport:

        三步:
        1. 知识图谱查询: task_description 中的概念 → kg.search() → 缺失列表
        2. 代码图谱查询: task_files 中的路径 → cg.query_children() → 未覆盖子模块
        3. 模糊决策点: 基于 heuristics 提取（文件跨模块引用、配置项、环境变量）
        """
        ...
    
    def merge_into_context(self, report: BlindspotReport) -> str:
        """将扫描报告转为可注入上下文的文本块。"""
        ...
```

### 3.3 MonitorAgent 集成

```python
# metacognition/monitor.py — MonitorAgent 新增方法
class MonitorAgent:
    async def preflight_blindspot(
        self, task_desc: str, task_files: list[str],
    ) -> BlindspotReport | None:
        """任务创建时执行盲区扫描。
        
        调用时机: Goal 创建后、Agent 开始执行前。
        跳过条件: GoalSession.blindspot_skipped == True
        """
        if self._blindspot_scanner is None:
            return None
        return await self._blindspot_scanner.scan(task_desc, task_files)
```

### 5.4 触发条件

```python
# 触发判断——仅核心模块改动
CORE_MODULES = [
    "scheduler/", "graph/", "hallucination/", "checkpoint/",
    "agents/", "gateway/", "compliance/", "evolution/",
    "metacognition/", "sandbox/", "memory/", "compose/",
    "communication/",
]

def should_trigger_quiz(goal: GoalSession, changed_files: list[str]) -> bool:
    """判断是否需要测验。"""
    if not goal.quiz_enabled:  # 用户可在 task config 关闭
        return False
    return any(
        any(f.startswith(mod) for mod in CORE_MODULES)
        for f in changed_files
    )
```

## 7. 调度器状态变更

当前调度器状态机（`scheduler/`）无需新增状态。
- 偏离日志在现有状态转换中记录（不改变转换逻辑）
- 盲区扫描在 `PARSING` 阶段前执行（新阶段 `PREFLIGHT` 可选，但建议不新增——在现有 `PARSING` 入口处同步调用）
- 测验模式在 `VERIFYING` → `COMPLETED` 转换间插入门禁检查

**建议**: 不新增状态机状态。PREFLIGHT 作为 PARSING 的前置钩子，测验作为 VERIFYING 的后置钩子。最小侵入。

## 9. 边界 Case 清单

| 类别 | 场景 | 预期行为 |
|------|------|---------|
| 偏离日志 | 任务执行中无任何偏离 | 生成空的 implementation-notes.md（"严格按计划执行，无偏离"） |
| 偏离日志 | 同一文件被多个 Agent 并发写偏离日志 | 每个 Agent 独立写 entries，flush 时按时间戳合并 |
| 偏离日志 | 偏离日志文件写入失败（磁盘满） | 降级到内存缓存，checkpoint save 时再试；checkpoint 也失败则记 Critical 日志 |
| 偏离日志 | 任务熔断时偏离日志未 flush | 熔断处理中先 flush_to_checkpoint() 再回滚 |
| 盲区扫描 | 知识图谱为空（新项目） | 跳过 knowledge_gaps 扫描，仅完成 code_gaps + fuzzy_decisions |
| 盲区扫描 | 代码图谱未初始化 | 全部降级为 fuzzy_decisions：提示用户手动指定关注文件 |
| 盲区扫描 | 扫描超时（>5 秒） | 返回已收集的部分结果 + warning 标记，不阻塞任务创建 |
| 盲区扫描 | task_description 为空 | 跳过扫描，返回空报告 |
| 测验 | Agent 任务失败/熔断 | 跳过测验——没有 diff 需要理解 |
| 测验 | LLM 生成测验失败（gateway 超时） | 降级：生成静态模板（"请手动审查以下 diff"），不阻塞完成 |
| 测验 | 用户连续答错 3 次 | 第 3 次失败后显示"建议重新审查 diff 或请求 code review" |
| 测验 | 非核心模块用户手动开启测验 | 正常生成测验，流程一致 |
| 跨模块 | GoalSession 缺 quiz_passed 字段（旧数据） | 默认 False，向后兼容 |

## 10. 风险与缓解

| 风险 | 概率 | 影响 | 缓解 |
|------|------|------|------|
| **偏离判断误分类**：minor 被标为 major，噪音进入 evolution | 中 | 低——evolution 信号稀释 | 偏离判定用白名单规则（数据模型/API/安全），不依赖 LLM |
| **盲区扫描性能**：知识图谱+代码图谱双查询超 5 秒 | 低 | 中——阻塞任务创建 | 两个查询并行执行（asyncio.gather），5 秒超时 abort |
| **测验 LLM 生成质量**：LLM 出的题太简单/太偏 | 中 | 中——门禁失效 | rubric.md 约束出题规范 + 未来可 A/B 对比人工出题质量 |
| **checkpoint PG 列变更**：ALTER TABLE 锁表 | 低 | 中——checkpoint 写操作短暂阻塞 | SQLite MVP 阶段无此风险；PG 用 `ADD COLUMN IF NOT EXISTS` |
