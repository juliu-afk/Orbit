### 4.4 Composable：useWebSocket

```
// composables/useWebSocket.ts
// WHY 抽成 composable 而非写在 store 里：
// WebSocket 连接生命周期独立于 Pinia，多个 store 消费同一连接。
// WHY 原生 WebSocket 而非 Socket.IO：
// 场景简单（订阅→推送），无需房间/多路复用重器。
// 浏览器 WebSocket API 已足够成熟，零依赖，重连行为完全可控。

export function useWebSocket() {
  const ws = ref<WebSocket | null>(null)
  const connectionStatus = ref<'connecting'|'connected'|'disconnected'>('disconnected')
  const retryCount = ref(0)
  const maxRetries = 5
  const currentTaskId = ref<string | null>(null)
  let reconnectTimer: ReturnType<typeof setTimeout> | null = null

  function connect(url: string) {
    // 清理旧连接
    if (ws.value) { ws.value.close(); ws.value = null }

    connectionStatus.value = 'connecting'
    ws.value = new WebSocket(url)

    ws.value.onopen = () => {
      connectionStatus.value = 'connected'
      retryCount.value = 0
      // 重连后自动恢复订阅
      if (currentTaskId.value) {
        ws.value!.send(JSON.stringify({
          type: 'subscribe',
          task_id: currentTaskId.value
        }))
      }
    }

    ws.value.onmessage = (event: MessageEvent) => {
      // WHY try-catch：后端推送格式异常（如字段缺失）时
      // 全局兜底，不让一条坏消息炸掉整个 Dashboard。
      try {
        const msg = JSON.parse(event.data)
        switch (msg.type) {
          case 'task:update':
            useTaskStore().handleTaskUpdate(msg.payload); break
          case 'token:update':
            useTokenData().addPoint(msg.payload); break
          case 'alert:new':
            useAlertStore().addAlert(msg.payload); break
        }
      } catch (e) {
        console.warn('[useWebSocket] 消息解析失败', e)
      }
    }

    ws.value.onerror = () => {
      // 不在此处重连——onclose 会在 error 后触发，统一在 onclose 处理
    }

    ws.value.onclose = (event: CloseEvent) => {
      connectionStatus.value = 'disconnected'
      ws.value = null
      // 非正常关闭 + 未达重试上限 → 指数退避重连
      if (!event.wasClean && retryCount.value < maxRetries) {
        scheduleReconnect(url)
      }
    }
  }

  function scheduleReconnect(url: string) {
    // 指数退避: 1s → 2s → 4s → 8s → 16s
    const delay = Math.min(1000 * Math.pow(2, retryCount.value), 16000)
    retryCount.value++
    reconnectTimer = setTimeout(() => connect(url), delay)
  }

  function subscribe(taskId: string) {
    currentTaskId.value = taskId
    if (ws.value && connectionStatus.value === 'connected') {
      ws.value.send(JSON.stringify({ type: 'subscribe', task_id: taskId }))
    }
  }

  function disconnect() {
    if (reconnectTimer) { clearTimeout(reconnectTimer); reconnectTimer = null }
    retryCount.value = maxRetries  // 禁止自动重连
    ws.value?.close(1000, 'client disconnect')
    ws.value = null
    connectionStatus.value = 'disconnected'
  }

  return {
    connectionStatus,
    retryCount,
    maxRetries,
    connect,
    disconnect,
    subscribe,
  }
}
```

### 5.1 EventBus 设计

```python
# src/orbit/events/bus.py
"""事件总线：asyncio.Queue 实现发布-订阅。

WHY asyncio.Queue 而非 Redis Pub/Sub：
MVP 单进程足够，零配置零延迟。多进程时迁移到 Redis 仅需换实现，
EventBus 接口不变。
"""

from __future__ import annotations

import asyncio
from typing import Any

from orbit.events.schemas import DashboardEvent


class EventBus:
    """进程内事件总线。

    调度器 publish，WS 广播器 subscribe（单消费者）。
    WHY 单消费者非多消费者：只有一个 WS 广播协程消费所有事件，
    避免事件被多个协程重复处理。
    """

    def __init__(self, maxsize: int = 256):
        self._queue: asyncio.Queue[DashboardEvent] = asyncio.Queue(maxsize=maxsize)

    def publish(self, event: DashboardEvent) -> None:
        """发布事件（非阻塞）。

        WHY put_nowait：调度器状态转换是同步的，阻塞会卡住状态机。
        队列满时丢弃并警告（Dashboard 容忍丢失，不丢数据一致性）。
        """
        try:
            self._queue.put_nowait(event)
        except asyncio.QueueFull:
            import structlog
            structlog.get_logger().warning("EventBus queue full, dropping event", type=event.type)

    async def subscribe(self) -> DashboardEvent:
        """异步消费事件（阻塞等待）。"""
        return await self._queue.get()
```

### 5.2 事件数据模型

```python
# src/orbit/events/schemas.py
"""Dashboard 事件数据模型。

WHY 独立于 API schemas：事件模型面向推送，字段精简；
API schemas 面向请求/响应，字段完整。职责不同。
"""

from __future__ import annotations

from datetime import datetime, timezone
from typing import Any, Literal

from pydantic import BaseModel, Field


class DashboardEvent(BaseModel):
    """事件基类。"""
    type: str
    task_id: str
    payload: dict[str, Any]
    timestamp: datetime = Field(default_factory=lambda: datetime.now(timezone.utc))


class TaskUpdatePayload(BaseModel):
    """task:update 事件 payload。"""
    task_id: str
    state: str                     # TaskState value
    progress: float
    dag: list[dict[str, Any]]      # 简化版 GraphNode 列表
    timestamp: str


class TokenUpdatePayload(BaseModel):
    """token:update 事件 payload。"""
    task_id: str
    prompt_tokens: int
    completion_tokens: int
    total_tokens: int
    timestamp: str


class AlertPayload(BaseModel):
    """alert:new 事件 payload。"""
    task_id: str
    level: str                     # HallucinationLevel value
    severity: Literal["warning", "critical"]
    message: str
    timestamp: str
```

### 5.3 WebSocket 服务端

```python
# src/orbit/ws/manager.py
"""WebSocket 连接管理器。

WHY 手写而非 Socket.IO room：原生 WebSocket 足够简单，
连接→订阅 task_id→推送，30 行代码替代一个第三方库依赖。
"""

from __future__ import annotations

import structlog
from fastapi import WebSocket

logger = structlog.get_logger()


class ConnectionManager:
    """WebSocket 连接 + 任务订阅管理。

    数据结构：{task_id: {WebSocket, ...}}
    一个 task_id 可被多个客户端订阅（多 Tab / 多运维）。
    """

    def __init__(self):
        # WHY set 而非 list：同一个 WS 对象不会重复出现在同一 task 房间，
        # 且 remove 是 O(1)（虽然 set 只能存 hashable，但 WebSocket 是 id-based）。
        self._rooms: dict[str, set[WebSocket]] = {}

    async def connect(self, ws: WebSocket) -> None:
        """接受连接。不做认证（PRD Non-Goal）。"""
        await ws.accept()

    async def disconnect(self, ws: WebSocket) -> None:
        """从所有房间移除。"""
        for task_id in list(self._rooms):
            self._rooms[task_id].discard(ws)
            if not self._rooms[task_id]:
                del self._rooms[task_id]

    async def subscribe(self, ws: WebSocket, task_id: str) -> None:
        """客户端订阅任务推送。"""
        self._rooms.setdefault(task_id, set()).add(ws)
        logger.info("ws_subscribe", task_id=task_id)

    async def unsubscribe(self, ws: WebSocket, task_id: str) -> None:
        """客户端取消订阅。"""
        if task_id in self._rooms:
            self._rooms[task_id].discard(ws)
            if not self._rooms[task_id]:
                del self._rooms[task_id]

    async def broadcast(self, task_id: str, data: dict) -> None:
        """向订阅某 task 的所有客户端推送。

        WHY 清理死连接：WebSocket 可能因网络中断未触发 disconnect，
        发送时 RuntimeError 表示连接已死，批量清理。
        """
        dead: set[WebSocket] = set()
        for ws in self._rooms.get(task_id, set()):
            try:
                await ws.send_json(data)
            except RuntimeError:
                dead.add(ws)
        if dead:
            self._rooms[task_id] -= dead
            if not self._rooms[task_id]:
                del self._rooms[task_id]
```

```python
# src/orbit/ws/router.py
"""WebSocket 端点 + 广播协程。

WHY WebSocket 路由写在独立文件：main.py 只组装，路由逻辑隔离。
"""

from __future__ import annotations

import asyncio
import json

import structlog
from fastapi import APIRouter, WebSocket, WebSocketDisconnect

from orbit.events.bus import EventBus
from orbit.ws.manager import ConnectionManager

logger = structlog.get_logger()

router = APIRouter()
manager = ConnectionManager()


async def start_broadcaster(bus: EventBus):
    """后台协程：消费 EventBus → 推送到订阅客户端。

    WHY 独立协程：避免调度器 publish() 直接 await ws.send_json()
    阻塞状态机。广播是 I/O 密集操作，放到后台。
    """
    while True:
        event = await bus.subscribe()
        await manager.broadcast(event.task_id, event.model_dump())


@router.websocket("/ws/dashboard")
async def dashboard_ws(websocket: WebSocket):
    """驾驶舱 WebSocket 端点。

    协议（JSON 帧）：
    - C→S: {"type": "subscribe", "task_id": "..."}
    - C→S: {"type": "unsubscribe", "task_id": "..."}
    - S→C: {"type": "task:update", "task_id": "...", "payload": {...}}
    - S→C: {"type": "token:update", "task_id": "...", "payload": {...}}
    - S→C: {"type": "alert:new", "task_id": "...", "payload": {...}}
    """
    await manager.connect(websocket)
    try:
        while True:
            raw = await websocket.receive_text()
            try:
                msg = json.loads(raw)
            except json.JSONDecodeError:
                await websocket.send_json({
                    "type": "error", "payload": {"message": "invalid JSON"}
                })
                continue

            msg_type = msg.get("type")
            task_id = msg.get("task_id", "")

            if msg_type == "subscribe" and task_id:
                await manager.subscribe(websocket, task_id)
            elif msg_type == "unsubscribe" and task_id:
                await manager.unsubscribe(websocket, task_id)
    except WebSocketDisconnect:
        await manager.disconnect(websocket)
```

- `{"type": "token:update", "task_id": "...", "payload": {...}}`
- `{"type": "alert:new", "task_id": "...", "payload": {...}}`

## 10. 风险点

| 风险 | 影响 | 缓解 | PRD 引用 |
|------|------|------|---------|
| vis-network >50 节点卡顿 | DAG 图不可用 | `physics: false` + `hierarchical` + 折叠已完成节点 | PRD+ADR 风险1 |
| ECharts 实时更新内存泄漏 | Token 图 OOM | `dataZoom` 窗口 + 100 点上限 + `dispose` | PRD+ADR 风险2 |
| 手写重连逻辑有 bug（指数退避互斥等） | 断线不恢复 | 单元测试覆盖重连次数上限 + disconnect 后不复重连 | §9.1 |
| 调度器改造引入性能回退 | 核心链路受影响 | EventBus publish 非阻塞（put_nowait），不影响状态转换 | §5.1 |
| 前端首次构建配置（Vite+Vue3+TS） | 项目初始化耗时 | 参考 Vite 官方模板，最小化配置 | — |
